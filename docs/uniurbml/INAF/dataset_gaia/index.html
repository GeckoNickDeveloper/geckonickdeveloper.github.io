Remove non-float data

cols_to_drop = []

for i in range(len(df.columns)):
    if pd.api.types.infer_dtype(df[df.columns[i]]) != 'floating':
        cols_to_drop.append(df.columns[i])

usable = df.drop(columns=cols_to_drop)

Remove nan values

def filter_nan_col(df, col):
    filtered_df = df.dropna(subset=[col])
    return filtered_df

def filter_nan(df, cols):
    filtered_df = df
    for col in cols:
        if col not in df.columns:
            print(f'Skipped [{col}]')
        else:
            filtered_df = filter_nan_col(filtered_df, col)
            print(f'Post filter [{col}]: {filtered_df.shape[0]}')
    return filtered_df
